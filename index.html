let balance = 0; let step = 0; let wins = 0; let lossStreak = 0; // 🔥 NEW: deep mode tracker let winIndex = 0; let base = 2.8; let ladder = []; function apply(){ base = Number(document.getElementById("baseTrade").value); balance = Number(document.getElementById("startBal").value); ladder = [ base, base * 1.55, base * 2.34, base * 3.5, base * 5.25, base * 7.87 ]; step = 0; wins = 0; lossStreak = 0; winIndex = 0; update(); } // ================= TRADE ================= function trade(){ return ladder[Math.min(step, ladder.length - 1)]; } // ================= LOSS ================= function loss(){ let t = trade(); balance -= t; step = Math.min(step + 1, ladder.length - 1); lossStreak++; wins = 0; // reset deep tracker on loss winIndex = 0; update(); } // ================= WIN ================= function win(){ let t = trade(); let ret = t * 0.92; balance += ret; // ================= NORMAL MODE ================= if(lossStreak < 5){ wins++; if(wins >= 2){ step = 0; wins = 0; } else{ step = Math.max(0, step - 1); } } // ================= DEEP MODE ================= else{ winIndex++; // 🔥 FIXED SEQUENCE (NO LADDER) if(winIndex === 1){ step = 5; // 22.04 } else if(winIndex === 2){ step = 4; // 14.70 } else if(winIndex === 3){ step = 2; // 6.55 step = 0; // RESET lossStreak = 0; winIndex = 0; } } update(); } // ================= UPDATE ================= function update(){ document.getElementById("bal").innerText = "$" + balance.toFixed(2); document.getElementById("trd").innerText = "$" + trade().toFixed(2); document.getElementById("step").innerText = step; document.getElementById("wins").innerText = wins; document.getElementById("loss").innerText = lossStreak; } update();
